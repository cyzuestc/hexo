```
List<CartItem> cartItemList toCartItemList(List<Map<String,Object>> mapList){
    List<CartItem> cartItemList = new ArrayList<CartItem>();
    for(Map map:mapList){
        CartItem cartItem = toCartItem(map);
        cartItemList.add(cartItem);
    }
    return cartItemList;
}
```
